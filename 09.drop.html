Dropping Collections

We can use the drop() method to drop collections in MongoDB.

The drop() method will take a callback function containing an error object, and also the result parameter which will return true if the collection was dropped successfully, or false if the drop was not successful.


      var MongoClient = require('mongodb').MongoClient;
      var url = 'mongodb://localhost:27017/mydb';

      MongoClient.connect(url, function(err, db) {
        if(err) throw err;
        db.collection('customers').drop(function(err, delOK) {
          if(err) throw err;
          if(delOK) console.log('Collection Dropped Successfully!');
          db.close();
        });
      });

Run the app.js file containing the code above using the following:


      node app.js

      // Result: Collection Dropped Successfully!

db.dropCollection

We can also use the db.dropCollection() method to delete collections.

This method takes two parameters: the name of the collection and a callback function:


      var MongoClient = require('mongodb').MongoClient;
      var url = "mongodb://localhost:27017/mydb";

      MongoClient.connect(url, function(err, db) {
        if (err) throw err;
        db.dropCollection("customers", function(err, delOK) {
          if (err) throw err;
          if (delOK) console.log("Collection Dropped Successfully!");
          db.close();
        });
      });

Run the app.js file containing the code above using the following:


      node app.js

      // Result: Collection Dropped Successfully!
